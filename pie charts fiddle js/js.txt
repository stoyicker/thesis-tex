google.setOnLoadCallback(drawChart);
function drawChart() {

  var data = google.visualization.arrayToDataTable([
    ['Reference presence', 'Amount'],
    ['Interested', 4],
    ['No response', 9],
    ['Not interested', 0]
  ]);

  var options = {
    pieSliceTextStyle: {
      color: 'white',
      font: 'Ubuntu',
      bold: true
    },
    pieSliceText: 'value',
        legend: {
textStyle: {
    fontName: "Ubuntu",
    bold: true
}
}
  };

  var chart = new google.visualization.PieChart(document.getElementById('donut_single'));
  chart.draw(data, options);
}
